# Quick Git Push Commands for public_html

# Step 1: Add your Git repository URL as remote
git remote add origin <YOUR-REPO-URL>
# Example: git remote add origin https://github.com/yourusername/trinity-oil-mills.git

# Step 2: Add all files
git add .

# Step 3: Commit
git commit -m "Initial commit - Port 3001, ready for deployment"

# Step 4: Set main branch
git branch -M main

# Step 5: Push to Git
git push -u origin main

# That's it! Your public_html folder is now on Git.

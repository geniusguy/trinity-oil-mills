# Step 1: Find and add Git to PATH

    $gitDir = Split-Path $gitPath.FullName
    $env:Path = "$gitDir;$env:Path"
    Write-Host "Git found and added to PATH" -ForegroundColor Green
} elseif (Test-Path "C:\Program Files\Git\cmd") {
    $env:Path = "C:\Program Files\Git\cmd;$env:Path"
    Write-Host "Git found and added to PATH" -ForegroundColor Green
}
